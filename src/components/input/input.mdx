# Input

import { Playground } from 'docz';
import { useColorMode } from 'theme-ui';
import Input from './ReactInput';
import Select from './ReactSelect';
import Textarea from './ReactTextarea';

## Input

### React example

```jsx
import React from 'react';
import classNames from 'classnames';
import { useColorMode } from 'theme-ui';

const Input = ({
  valid = false,
  invalid = false,
  disabled = false,
  className,
  placeholder,
}) => {
  const [colorMode] = useColorMode();
  return (
    <input
      disabled={disabled}
      className={classNames({
        input: true,
        [className]: true,
        [`${colorMode}:input`]: true,
        [`${colorMode}:input-valid`]: valid,
        [`${colorMode}:input-invalid`]: invalid,
      })}
      placeholder={placeholder}
    />
  );
};
```

<Playground>
  <div className="max-w-sm">
    <Input placeholder="Regular" className="mb-4" />
    <Input placeholder="Invalid" invalid className="mb-4" />
    <Input placeholder="Valid" valid className="mb-4" />
    <Input placeholder="Disabled" disabled className="mb-4" />
  </div>
</Playground>
