# Card

import { Playground } from 'docz';
import { useColorMode } from 'theme-ui';
import Card from './ReactCard';

```jsx
const Card = ({
  colorMode = 'light',
  elevation = 1,
  className,
  children,
  ...rest
}) => (
  <div
    className={classNames(
      'card',
      `${colorMode}:card-elevation-${elevation}`,
      className
    )}
    {...rest}
  >
    {children}
  </div>
);
```
