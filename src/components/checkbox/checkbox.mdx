## Standalone checkbox

### React snippet

```jsx
import React from 'react';
import classNames from 'classnames';
import { useColorMode } from 'theme-ui';

const Checkbox = ({
  invalid = false,
  disabled = false,
  className,
  checked,
}) => {
  const [colorMode] = useColorMode();
  return (
    <input
      type="checkbox"
      checked={checked}
      disabled={disabled}
      className={classNames({
        checkbox: true,
        [className]: !!className,
        [`${colorMode}:checkbox-normal`]: !invalid,
        [`${colorMode}:checkbox-invalid`]: invalid,
      })}
    />
  );
};
```
