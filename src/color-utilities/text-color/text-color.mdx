# Text color

import { Playground } from 'docz';
import { useColorMode } from 'theme-ui';
import { colors } from '../../colors';
import { ClassNamesTable } from '../../docz-components';
import { mapNestedThemeToClasses } from '../../helpers';

<ClassNamesTable
  data={mapNestedThemeToClasses(colors, (colorMode, variant, num) => ({
    [`${colorMode}:text-${variant}-${num}`]: [
      `color: ${colors[colorMode][variant][num]}`,
    ],
  }))}
/>

<Playground>
  {() => {
    const [colorMode] = useColorMode();
    return Object.keys(colors[colorMode]).map((variant) => (
      <div key={variant} className="mb-12">
        {Object.keys(colors[colorMode][variant]).map((num) => (
          <p
            key={`${colorMode}-${variant}-${num}`}
            className={`${colorMode}:text-${variant}-${num}`}
          >
            The spectacle before us was sublime.
          </p>
        ))}
      </div>
    ));
  }}
</Playground>
